function renderSchedule(type) {
    const root = document.getElementById("schedule");
    root.innerHTML = "";

    Object.entries(scheduleData[type]).forEach(([day, lessons]) => {
        const div = document.createElement("div");
        div.className = "day fade";

        div.innerHTML = `<h2>${day}</h2>`;

        let pairs = {};

        lessons.forEach(item => {
            if (!pairs[item.pair]) pairs[item.pair] = [];
            pairs[item.pair].push(item);
        });

        Object.keys(pairs).sort((a,b)=>a-b).forEach(pairNum => {
            const block = document.createElement("div");
            block.className = "pair-block";

            block.innerHTML = `<div class="pair-title">${pairNum}-та пара</div>`;

            pairs[pairNum].forEach(item => {
                const s = document.createElement("div");
                s.className = `subject ${item.type}`;
                s.innerHTML = `
                    <b>${item.subject}</b><br>
                    <span>${item.teacher}</span><br>
                    <span>${item.group}</span>
                `;
                block.appendChild(s);
            });

            div.appendChild(block);
        });

        root.appendChild(div);
    });
}

document.getElementById("oddBtn").onclick = () => {
    oddBtn.classList.add("active");

    renderSchedule("odd");

document.getElementById("evenBtn").onclick = () => {
    evenBtn.classList.add("active");
    oddBtn.classList.remove("active");
